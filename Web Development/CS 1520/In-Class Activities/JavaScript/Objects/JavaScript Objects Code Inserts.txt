    function TV(brand, size, price, injacks, outjacks) {
        // object fields
        this.brand = brand;
        this.size = size;
        this.price = price;
        this.jacks = new Object();  // a field can also be an object
        this.jacks.input = injacks;
        this.jacks.output = outjacks;
        this.display = show_properties;  // assigning a function to the display field
    }

    function show_properties() {
        console.log("This is the TV:");
        console.log(`Brand: ${this.brand}`);
        console.log(`Size: ${this.size}`);
        console.log(`Price: ${this.price}`);
        console.log("Input Jacks:");
        console.log(`${this.jacks.input}`);
        console.log("Output Jacks:");
        console.log(`${this.jacks.output}`);
        console.log("\n");
    }

    function block5() {
        console.log("block #5");
        let tvs_in_my_house = [my_tv, your_tv, her_tv];

        // sorting array of TVs
        // sort() function: will it sort the TVs by what? size or price?
        // sorting by price
        console.log("sorting by price");
        tvs_in_my_house.sort(by_price);
        for (let current_tv of tvs_in_my_house) {
            current_tv.display();
        }

        // sorting by size
        console.log("sorting by size");
        tvs_in_my_house.sort(by_size);
        for (let current_tv of tvs_in_my_house) {
            current_tv.display();
        }
    }

    function by_size(tv1, tv2) {
        return tv1.size - tv2.size;
    }

    function by_price(tv1, tv2) {
        return tv1.price - tv2.price;
    }

    function block6() {
        console.log("block #6");
        // creating objects from JavaScript Classes
        let person01 = new Person("Joanna", 21);
        let person02 = new Person("Robert", 19);
        let person03 = new Person("Briana", 20);

        let people = [person01, person02, person03];
        for (let currentPerson of people) {
            currentPerson.display();
        }
    }

    // defining a new class named Person
    class Person {
        // class constructor and its fields name and age
        constructor(name, age) {
            this.name = name;
            this.age = age;
        }

        // class method
        display() {
            console.log("Name: " + this.name);
            console.log("Age: " + this.age + "\n");
        }
    }
